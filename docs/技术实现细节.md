```python
from PySide6.QtWidgets import QOpenGLWidget
from PySide6.QtOpenGL import QOpenGLFunctions
from PySide6.QtCore import Qt, QTimer, Signal

class Live2DWidget(QOpenGLWidget):
    """Live2D 渲染控件"""

    # 信号
    clicked = Signal()          # 点击信号
    drag_started = Signal()     # 拖拽开始
    drag_ended = Signal()       # 拖拽结束

    def __init__(self, parent=None):
        super().__init__(parent)
        self.renderer = None
        self.model = None
        self.motion_controller = None
        self.is_dragging = False
        self.last_pos = None

        # 配置透明背景
        self.setAttribute(Qt.WA_TranslucentBackground)
        self.setMouseTracking(True)

        # 渲染定时器 (60fps)
        self.render_timer = QTimer()
        self.render_timer.timeout.connect(self.update)
        self.render_timer.start(16)  # ~60fps

    def initializeGL(self):
        """OpenGL 初始化"""
        self.gl = QOpenGLFunctions()
        self.gl.initializeOpenGLFunctions()

        # 初始化渲染器
        self.renderer = Live2DRenderer(self.gl)
        self.renderer.initialize()

    def paintGL(self):
        """渲染一帧"""
        self.gl.glClearColor(0, 0, 0, 0)
        self.gl.glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT)

        if self.model:
            self.motion_controller.update()
            self.renderer.draw(self.model)

    def load_model(self, model_path: str):
        """加载 Live2D 模型"""
        self.model = Live2DModel.load(model_path)
        self.motion_controller = MotionController(self.model)

    # 鼠标事件处理
    def mousePressEvent(self, event):
        if event.button() == Qt.LeftButton:
            self.clicked.emit()
            if self.motion_controller:
                self.motion_controller.trigger_random_motion()

    def mouseMoveEvent(self, event):
        # 处理拖拽
        pass
```

```python
from PySide6.QtWidgets import (QMainWindow, QWidget, QHBoxLayout,

class MainWindow(QMainWindow):
    """主窗口"""

        super().__init__()
        self.setup_ui()
        self.setup_window_flags()
        self.drag_position = None

    def setup_ui(self):
        """设置 UI"""
        # 中央 widget
        central = QWidget()

        # 水平布局：[Live2D] + [对话窗口]
        self.layout = QHBoxLayout(central)

        self.layout.setSpacing(0)

        # Live2D 控件
        self.live2d_widget = Live2DWidget()
        self.live2d_widget.setFixedSize(300, 400)
        self.layout.addWidget(self.live2d_widget)

        # 对话窗口 (初始隐藏)
        self.chat_widget = ChatWidget()
        self.chat_widget.setVisible(False)
        self.chat_widget.setFixedWidth(400)
        self.layout.addWidget(self.chat_widget)

        # 对话切换按钮 (悬浮在 Live2D 上)
        self.toggle_button = QPushButton("💬", self.live2d_widget)
        self.toggle_button.setGeometry(250, 350, 40, 40)
        self.toggle_button.clicked.connect(self.toggle_chat)

    def setup_window_flags(self):
        """设置窗口标志"""
        self.setWindowFlags(
            Qt.FramelessWindowHint |  # 无边框
            Qt.WindowStaysOnTopHint | # 置顶
            Qt.Tool                   # 任务栏不显示
        )
        self.setAttribute(Qt.WA_TranslucentBackground)

    def toggle_chat(self):
        """切换对话窗口显示/隐藏"""
        visible = not self.chat_widget.isVisible()
        self.chat_widget.setVisible(visible)

    # 拖拽支持
    def mousePressEvent(self, event):
        if event.button() == Qt.LeftButton:
            self.drag_position = event.globalPosition().toPoint() - self.frameGeometry().topLeft()

    def mouseMoveEvent(self, event):
        if self.drag_position and event.buttons() == Qt.LeftButton:
            self.move(event.globalPosition().toPoint() - self.drag_position)
```

#### 1.3.1 LLM 客户端

```python
# src/chat/llm_client.py
from litellm import completion, acompletion
from typing import List, Dict, Any, Optional
from dataclasses import dataclass

@dataclass
class LLMConfig:
    provider: str  # "openai", "anthropic", "ollama"
    model: str
    api_key: Optional[str] = None
    base_url: Optional[str] = None
    temperature: float = 0.7
    max_tokens: int = 1000

class LLMClient:
    """LLM 客户端抽象"""

    def __init__(self, config: LLMConfig):
        self.config = config

    def _get_litellm_params(self) -> Dict[str, Any]:
        """获取 LiteLLM 参数"""
        params = {
            "model": self._map_model(),
            "temperature": self.config.temperature,
            "max_tokens": self.config.max_tokens,
        }
        if self.config.api_key:
            params["api_key"] = self.config.api_key
        if self.config.base_url:
            params["api_base"] = self.config.base_url
        return params

    def _map_model(self) -> str:
        """映射模型名称到 LiteLLM 格式"""
        if self.config.provider == "openai":
            return f"openai/{self.config.model}"
        elif self.config.provider == "anthropic":
            return f"anthropic/{self.config.model}"
        elif self.config.provider == "ollama":
            return f"ollama/{self.config.model}"
        return self.config.model

    async def chat(
        self,
        messages: List[Dict[str, str]],
        tools: Optional[List[Dict]] = None,
        stream: bool = False
    ) -> str:
        """对话"""
        params = self._get_litellm_params()
        params["messages"] = messages
        if tools:
            params["tools"] = tools

        if stream:
            return await self._chat_stream(params)
        else:
            response = await acompletion(**params)
            return response.choices[0].message.content

    async def _chat_stream(self, params: Dict):
        """流式输出"""
        response = await acompletion(stream=True, **params)
        async for chunk in response:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content
```

```python
from PySide6.QtWidgets import (QWidget, QVBoxLayout, QTextEdit,

from PySide6.QtCore import Qt, Signal, QThread

class MessageBubble(QFrame):

    def __init__(self, text: str, is_user: bool = False, parent=None):
        super().__init__(parent)
        self.setup_ui(text, is_user)

    def setup_ui(self, text: str, is_user: bool):

        label = QLabel(text)
        label.setWordWrap(True)
        label.setTextFormat(Qt.MarkdownText)
        layout.addWidget(label)

        # 样式
        if is_user:
            self.setStyleSheet("""

                QLabel { color: white; }
            """)
        else:
            self.setStyleSheet("""
                QFrame { background-color: #E9E9EB; border-radius: 15px; padding: 10px; }
                QLabel { color: black; }
            """)

class ChatWidget(QWidget):
    """对话窗口"""

    send_message = Signal(str)

    def __init__(self, parent=None):
        super().__init__(parent)
        self.setup_ui()

    def setup_ui(self):
        layout = QVBoxLayout(self)

        # 消息列表滚动区
        scroll = QScrollArea()
        scroll.setWidgetResizable(True)
        scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarAlwaysOff)

        self.messages_container = QWidget()
        self.messages_layout = QVBoxLayout(self.messages_container)
        self.messages_layout.addStretch()
        scroll.setWidget(self.messages_container)
        layout.addWidget(scroll)

        # 输入区
        input_layout = QHBoxLayout()
        self.input_edit = QLineEdit()
        self.input_edit.setPlaceholderText("输入消息...")
        self.input_edit.returnPressed.connect(self.on_send)

        send_btn = QPushButton("发送")
        send_btn.clicked.connect(self.on_send)

        input_layout.addWidget(self.input_edit)
        input_layout.addWidget(send_btn)
        layout.addLayout(input_layout)

    def on_send(self):
        text = self.input_edit.text().strip()
        if text:
            self.send_message.emit(text)
            self.add_message(text, is_user=True)
            self.input_edit.clear()

    def add_message(self, text: str, is_user: bool = False):
        """添加消息"""
        bubble = MessageBubble(text, is_user)
        self.messages_layout.insertWidget(self.messages_layout.count() - 1, bubble)
```

```python
# src/scheduler/parser.py
from typing import Optional, Dict, Any
import json

class TaskParser:
    """LLM 驱动的任务解析器"""

    SYSTEM_PROMPT = """你是一个定时任务解析器。将用户的自然语言描述转换为定时任务配置。

Cron 表达式格式：分 时 日 月 周
- * 表示任意值
- , 分隔多个值
- - 表示范围
- / 表示间隔

只返回 JSON，不要其他文字。"""

            "name": "create_scheduled_task",
            "description": "创建定时任务",
            "parameters": {
                "type": "object",
                "properties": {
                    "cron": {
                        "type": "string",
                        "description": "Cron 表达式"
                    },
                    "task_name": {
                        "type": "string",
                        "description": "简短的任务名称"
                    },
                    "action_prompt": {
                        "type": "string",
                        "description": "任务执行时发送给 LLM 的完整 prompt"
                    },
                    "motion_id": {
                        "type": "string",
                        "description": "可选，Live2D 动作 ID"
                    }
                },
                "required": ["cron", "task_name", "action_prompt"]
            }
        }
    }]

    def __init__(self, llm_client):
        self.llm_client = llm_client

    async def parse(self, user_input: str) -> Optional[Dict[str, Any]]:
        """解析用户输入为任务配置"""
        messages = [
            {"role": "system", "content": self.SYSTEM_PROMPT},
            {"role": "user", "content": user_input}
        ]

        response = await self.llm_client.chat(
            messages,
            tools=self.TOOLS
        )

        # 解析 tool_call
        if hasattr(response, 'choices') and response.choices[0].message.tool_calls:
            tool_call = response.choices[0].message.tool_calls[0]
            return json.loads(tool_call.function.arguments)
        return None
```

#### 1.4.3 任务管理器

```python
# src/scheduler/manager.py
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from apscheduler.triggers.cron import CronTrigger
from typing import Callable, Optional
import json
from pathlib import Path
from .task import ScheduledTask

class TaskManager:
    """定时任务管理器"""

    def __init__(self, storage_path: Path, on_task_execute: Callable):
        self.storage_path = storage_path
        self.on_task_execute = on_task_execute
        self.scheduler = AsyncIOScheduler()
        self.tasks: dict[str, ScheduledTask] = {}

    async def start(self):
        """启动调度器"""
        await self._load_tasks()
        self.scheduler.start()
        for task in self.tasks.values():
            if task.enabled:
                self._add_job(task)

    async def add_task(self, task: ScheduledTask) -> bool:
        """添加任务"""
        self.tasks[task.id] = task
        if task.enabled:
            self._add_job(task)
        await self._save_tasks()
        return True

    def _add_job(self, task: ScheduledTask):
        """添加 APScheduler job"""
        parts = task.cron.split()
        if len(parts) != 5:
            raise ValueError("Invalid cron expression")

        minute, hour, day, month, day_of_week = parts

        self.scheduler.add_job(
            self._execute_task,
            trigger=CronTrigger(
                minute=minute,
                hour=hour,
                day=day,
                month=month,
                day_of_week=day_of_week
            ),
            args=[task.id],
            id=task.id,
            replace_existing=True
        )

    async def _execute_task(self, task_id: str):
        """执行任务"""
        task = self.tasks.get(task_id)
        if not task:
            return

        task.last_run = datetime.utcnow()
        await self._save_tasks()

        # 回调执行
        await self.on_task_execute(task)

    async def _load_tasks(self):
        """从文件加载任务"""
        if self.storage_path.exists():
            data = json.loads(self.storage_path.read_text())
            for task_data in data.get("tasks", []):
                task = ScheduledTask(**task_data)
                self.tasks[task.id] = task

    async def _save_tasks(self):
        """保存任务到文件"""
        data = {
            "tasks": [t.dict() for t in self.tasks.values()]
        }
        self.storage_path.write_text(json.dumps(data, indent=2, ensure_ascii=False))
```

```python
# src/assistant/tool_registry.py
from typing import Dict, List, Callable, Any
from dataclasses import dataclass

@dataclass
class Tool:
    """工具定义"""
    name: str
    description: str
    parameters: Dict[str, Any]
    handler: Callable

class ToolRegistry:
    """工具注册表"""

    def __init__(self):
        self._tools: Dict[str, Tool] = {}

    def register(self, tool: Tool):
        """注册工具"""
        self._tools[tool.name] = tool

    def get_tool(self, name: str) -> Optional[Tool]:
        """获取工具"""
        return self._tools.get(name)

    def to_openai_tools(self) -> List[Dict]:
        """转换为 OpenAI tools 格式"""
        return [
            {
                "type": "function",
                "function": {
                    "name": tool.name,
                    "description": tool.description,
                    "parameters": tool.parameters
                }
            }
            for tool in self._tools.values()
        ]

    async def execute(self, name: str, arguments: Dict) -> Any:
        """执行工具"""
        tool = self._tools.get(name)
        if not tool:
            raise ValueError(f"Tool not found: {name}")
        return await tool.handler(**arguments)
```
